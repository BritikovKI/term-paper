\section{Preliminaries}
\label{Sec:Preliminaries}
This section introduces the definitions and formalisms used in the paper. 
Logical formulas in our approach are defined over the theory of Linear Integer Arithmetic (LIA).
We use $x,x'$ to denote set of variables (unprimed and primed), where $x' = \{v'| v \in x\}$, $x$ is state variables, and $x'$ is next state variables.
If there are variables of many different states in the formula we use superscript notation, e.g., $x^{(0)}, \dots, x^{(n)}$ .
We use $s, s_1, s_2 \dots$ to denote the specific states in the system.

%

The formula defined over $x$ is a state formula, and a formula over $x \cup x'$ is a transition formula.
State formulas represent a set of states where they hold, and transition formulas define binary relations over the state.
Using $Id(x,x')$ we denote a transition formula $x = x'$

\paragraph{Transition systems and Safety Problem} 
A \emph{transition system} (TS) is defined as $T = \tuple{ Init, Tr, x }$, where $Init$ is a formula encoding the initial states of the system, $Tr$ is a transition formula, which encodes the transition relation, and $x$ is a set of system's variables.
With respect to transition system, a \emph{transition trace} is a formula encoding multiple consecutive transitions: $Tr^{n}(x^{(0)}, \dots, x^{(n)}) = Tr(x^{(0)}, x^{(1)}) \land \dots \land Tr(x^{(n-1)}, x^{(n)})$.
% $trace(s_0, \dots, s_n) = Tr(s_0, s_1)) \land \dots \land Tr(s_{n-1}, s_n)$
We write $Tr^{n}(x,\dots,x')$ for this conjunction throughout the paper.
A \emph{safety problem} is defined as  $P = \tuple{ Init, Tr, Bad, x }$ where $Bad$ is a formula that represents states that violate the safety property.

%

Safety verification amounts to determining whether there exists a transition trace from the initial states that reaches a $Bad$ state.
If TS is safe, then there exists an inductive invariant $Inv(x)$ s.t. (1) $\forall x. Init(x) \rightarrow Inv(x)$, (2) $\forall x. Inv(x) \land Tr(x,x') \rightarrow Inv(x')$, and (3) $\forall x. Inv(x) \rightarrow \lnot Bad(x)$.
We use $safetyCheck(P)$ to denote a safety verification procedure that returns a tuple $\tuple{SAFE, Inv(x)}$ if TS is safe ($Inv(x)$ is inductive invariant).
Otherwise, if TS is unsafe it returns $\tuple{UNSAFE, Tr^{n}(x^{(0)}, \dots, x^{(n)})}$, such that: 
$$\exists x^{(0)}, \dots, x^{(n)}. Init(x^{(0)}) \land Tr^{n}(x^{(0)}, \dots, x^{(n)}) \land Bad(x^{(n)}).$$

\paragraph{Nontermination}
A transition system is called \textbf{nonterminating} if there exists an initial state, from which it is possible to take any number of transitions:
$$\exists x. Init(x) \land \forall n \in \mathbb{N}.  \exists x^{(1)}, \dots, x^{(n)}. Tr^{n}(x, x^{(1)}, \dots, x^{(n)}).$$

\begin{definition}[Recurrent Set]
    For a Transition System $\tuple{ Init, Tr, x }$, a formula $N(x)$ is a recurrent set if: (1) $\forall x. N(x) \rightarrow \exists x'. Tr(x,x') \land N(x')$,  (2) $\exists x. N(x) \land Init(x)$.
\end{definition}

Transition System is nonterminating if there exists a \emph{recurrent set}. 
Intuitively, every state in the recurrent set has a successor that also lies in the recurrent set, and the recurrent set contains some of the 
initial states.

\paragraph{Termination} A TS is called \textbf{terminating} if for every initial state it is possible to take only limited number of transitions.
For a Transition System $\tuple{ Init, Tr, x }$:
$$\forall x. Init(x) \rightarrow \exists n \in \mathbb{N}. \lnot \exists x^{(1)}, \dots, x^{(n)}. Tr^{n}(x, x^{(1)}, \dots, x^{(n)}).$$
Termination can be proven by synthesizing a \emph{ranking function}~\cite{DBLP:conf/vmcai/PodelskiR04}, mapping states to a well-founded domain, such that its value strictly decreases along every transition.

% Only relation can be well-founded, not formula!

\begin{definition}[Transition invariant]
    For a Transition System $\tuple{ Init, Tr, x }$, $TrInv(x,x')$ is a transition invariant if: $\forall x,x'. R(x) \land Tr^*(x,x') \rightarrow TrInv(x,x')$, where $Tr^*(x,x')$ is the reflexive transitive closure of $Tr(x,x')$
    and $R(x)$ is a formula encoding all the states, reachable within transition system.
\end{definition}

\emph{Transition invariant}~\cite{DBLP:conf/lics/PodelskiR04} can be used to prove termination.
If transition invariant is a finite union of well-founded relations (for example if every separate relation has a ranking function), 
then the Transition System terminates.
\begin{definition}[Disjunctively well-founded transition invariant]
    A transition invariant $TrInv(x,x') = \cup_{i = 0}^{n} R_i(x,x')$ is called disjunctively well-founded if every relation $R_i(x,x')$ is well-founded. 
    % A transition invariant $TrInv(x,x') = \bigvee_{i = 0}^{n} R_i(x,x')$ is called disjunctively well-founded if every disjunct $R_i(x,x')$ is a well-founded relation. 
\end{definition}

\paragraph{Craig Interpolation} A technique for computing an overapproximation of the formula~\cite{Craig_1957}.
Particularly, given two formulas $A$ and $B$ such that $A \land B$ is unsatisfiable, a Craig interpolant for $A$ and $B$ is a formula $I$ such that $A \rightarrow I$, $I \land B$ is unsatisfiable, and $I$ only contains the symbols, common to both $A$ and $B$.
We use $Itp(A, B)$ to denote the computation of the Craig interpolant for $A$ and $B$.

\paragraph{Proving Nontermination via Safety}
Many techniques reduce termination and nontermination analysis to safety verification.
One notable approach is the reduction of nontermination to safety, which constructs strong recurrent sets via inductive invariants~\cite{DBLP:conf/tacas/ChenCFNO14}.
It was introduced in the context of proving nontermination of programs with loops, and can be extended for the concept of Transition Systems.
This extension is described below.

\begin{definition}[Sink states]
    For a Transition System $\tuple{ Init, Tr, x }$, a state $s$ is a sink state, if $\lnot \exists s'. Tr(s,s')$.
    We use $Sink(x)$ to denote the formula encoding all sink states in the transition system, such that $\forall x. Sink(x) \rightarrow \lnot \exists x'. Tr(x,x')$
\end{definition}

The Algorithm~\ref{Alg:SNA} for Safety-based Nontermination analysis takes a safety problem $P = \tuple{Init, Tr, Sink, x}$, where $Sink(x) = QE(\lnot Tr(x,x'). \tuple{x'})$.
As an output, it returns NONTERM if the system is nonterminating, and UNKNOWN, if the algorithm cannot conclude the analysis. 
The Figure~\ref{fig:SNA} contains the execution flow of Safety-based Nontermination analysis.


\begin{algorithm}[t!]
	\SetInd{0.5em}{0.5em}
    \DontPrintSemicolon
    \small
    \newcommand{\Continue}{\textbf{continue}}
    \caption{Nontermination analysis using Safety.~\cite{DBLP:conf/tacas/ChenCFNO14}}
    \label{Alg:SNA}
    \SetKwInOut{Input}{Input}
    \SetKwInOut{Output}{Output}
    \Input{Safety problem $P = \tuple{Init, Tr, Sink, x}$}
    \Output{NONTERM | UNKNOWN}
    % \SetKwFunction{FAnalyze}{Analyze}
    % \SetKwProg{Fn}{Function}{:}{}
    % \nonl\Fn{\FAnalyze{$P$}}{
        \While{$\top$}{
            $\tuple{res,\phi = Tr^{n}\ or\ Inv} \gets safetyCheck(P)$\\
            \label{SNA:UNSAFE}\eIf{$\tuple{res,\phi} = \tuple{UNSAFE, Tr^{n}}$} {
                $P \gets \blockTerm(P, Tr^{n})$ \label{SNA:BLOCKTERM}
            }
            {
                \label{SNA:SAFE}
                \If{$\forall x. Inv(x) \rightarrow \exists x'. Tr(x,x') \land Inv(x')$} {
                    \Return NONTERM \label{SNA:RECURRENT}
                }
                \Return UNKNOWN
            }
            
        }
    % }
    
\end{algorithm}

The approach executes a safety check of $P$ in a loop.
At line~\ref{SNA:UNSAFE}, the algorithm handles the case when $P$ is unsafe (i.e., a trace reaching a sink state exists).
In this case, the algorithm calls the $\blockTerm$ procedure (Algorithm~\ref{Alg:SNABlock}), which refines the safety problem $P$ by blocking the states that are determenistically
leading to the sink states, updating the Safety problem.
$\blockTerm$ is described in detail below.
In the case when $P$ is safe, checks if inductive invariant $Inv(x)$ is a recurrent set (line~\ref{SNA:SAFE}).
The algorithm checks whether $Inv(x)$ is a recurrent set at line~\ref{SNA:RECURRENT}, meaning that it is possible to take a transition from every state within the invariant to another state.
If $Inv(x)$ is a recurrent set, then the transition system is nonterminating.
Otherwise, algorithm returns UNKNOWN, as the analysis is inconclusive.

\begin{algorithm}
    \SetInd{0.5em}{0.5em}
    \DontPrintSemicolon
    \small
    \newcommand{\Continue}{\textbf{continue}}
    \caption{\blockTerm(P, $Tr^{n}$).~\cite{DBLP:conf/tacas/ChenCFNO14}}
    \label{Alg:SNABlock}
    \SetKwInOut{Input}{Input}
    \SetKwInOut{Output}{Output}
    \Input{Safety problem $P = \tuple{Init, Tr, Sink, x}$, terminating trace $Tr^{n}(x^{(0)}, \dots, x^{(n)}) = Tr(x^{(0)}, x^{(1)}) \land \dots \land Tr(x^{(n-1)}, x^{(n)})$}
    \Output{Refined safety problem $P' = \tuple{Init', Tr', Sink', x}$}

    % \SetKwFunction{FBlock}{BlockTerm}
    % \SetKwProg{Fn}{Function}{:}{}
    % \Fn{\FBlock{$P, trace$}}{
    \SetAlgoFuncName{BlockTerm}{autoref name}
        $i \gets n-1$ \\
        \While{$i > 0$}{
            $Next(x^{(i+1)}) \gets QE(Init(x^{(0)}) \land Tr^{n} \land Sink(x^{(n)}), \tuple{x^{(0)},\dots, x^{(i)},  x^{(i+2)}, \dots})$ \\
            $Curr(x^{(i)}) \gets QE(Init(x^{(0)}) \land Tr^{n} \land Sink(x^{(n)}), \tuple{\exists x^{(0)},\dots, x^{(i-1)},  x^{(i+1)}, \dots })$ \\
            % $s_i(x) \gets $ \textit{variable assigment extracted from } $M$ for $x^{(i)}$\\
            % $s_{i+1}(x) \gets $ \textit{variable assigment extracted from } $M$ for $x^{(i+1)}$\\
            \label{SNA:BLOCK} \If{SAT ? $Curr(x) \land Tr(x,x') \land \lnot Next(x')$}{
                $Tr'(x,x') \gets Tr(x,x') \land \lnot Next(x')$ \label{SNA:TRBLOCK}  \\
                \Return $\tuple{Init, Tr', Sink, x}$
            }
            $i \gets i - 1$   
        }
        $Init'(x^{(0)}) \gets Init(x^{(0)}) \land \lnot QE(\exists x^{(1)}, \dots, x^{(n)} | Init(x^{(0)}) \land Tr^{n} \land Sink(x^{(n)}))$  \label{SNA:BLOCKINIT} \\
        \Return $\tuple{Init', Tr, Sink, x}$
    % }
\end{algorithm}

The $\blockTerm$ procedure takes as input a safety problem $P$ and a trace formula $Tr^{n}(x^{(0)}, \dots, x^{(n)})$ such that $Init(x) \land Tr^{n}(x, \dots, x') \land Sink(x')$ is satisfiable.
The algorithm iteratively checks if it is possible to take a transition that does not lead to the sink states(at line~\ref{SNA:BLOCK}), using a query to the SMT solver.
If it is possible, then the algorithm blocks the states that deterministically lead to the sink states at line~\ref{SNA:TRBLOCK}.
Otherwise, if such states do not exist, the algorithm blocks the subset of initial states that deterministically lead to the sink states (line ~\ref{SNA:BLOCKINIT}).


\begin{lemma}\label{Lem:NONTERM}
    If Algorithm~\ref{Alg:SNA} returns NONTERM, then the transition system is nonterminating.
\end{lemma}
